realtimeQuote

  • realtimeQuote(symbols: string[]): Promise<RealtimeQuote[]>

  • Get real-time quote

    Example

     
    import { Config, QuoteContext, SubType } from 'longbridge'

    let config = Config.fromEnv()
    let ctx = new QuoteContext(config);
    await ctx.open()

        ctx.realtimeQuote(["700.HK", "AAPL.US"]).then(resp => {
